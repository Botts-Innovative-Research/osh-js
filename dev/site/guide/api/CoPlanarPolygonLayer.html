Class: CoPlanarPolygonLayer

Example
import CoPlanarPolygonLayer from 'core/ui/layer/PolygonLayer.js';

 let boundedDrapingLayer = new PolygonLayer({
           getVertices : {
                dataSourceIds: [boundsDatasource.getId()],
                handler: (rec) => {
                    return [
                        rec.ulc.lon,
                        rec.ulc.lat,
                        rec.ulc.alt,
                        rec.urc.lon,
                        rec.urc.lat,
                        rec.ulc.alt,
                        rec.llc.lon,
                        rec.llc.lat,
                        rec.ulc.alt,
                        rec.lrc.lon,
                        rec.lrc.lat,
                        rec.ulc.alt,
                    ];
                }
            },
    });
